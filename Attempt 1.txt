void solve()
{
    ll n,a,b,c;
    cin>>n>>a>>b>>c;
    a-=c;
    b-=c;
    if(a<0 || b<0 || ((a+b+c)>n))
    {
        cout<<"IMPOSSIBLE";
        return;
    }
    ll k=n;
    ll A[n];
    for(ll i=0;i<a;i++)
        A[i]=n-1;
    for(ll i=a;i<a+c;i++)
        A[i]=n;
    ll t=n-a-b-c;
    for(ll i=a+c;i<a+c+t;i++)
        A[i]=1;
    for(ll i=a+c+t;i<n;i++)
        A[i]=n-1;
    for(ll i=0;i<n;i++)
        cout<<A[i]<<" ";
}

    for(int z=1;z<=t;z++)
    {
        cout<<"Case #"<<z<<": ";
        solve();
        cout<<en;
    }
}
